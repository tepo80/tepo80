          python-version: 3.11
      - run: pip install requests psutil retrying PyYaml || echo "skip install"
      - run: |
          rm -f final1.json normal1.json
          timeout 600s python cl1.py || echo "❌ cl1.py failed or skipped"
      - run: |
          if [[ -s final1.json || -s normal1.json ]]; then
            git config user.name "github-actions[bot]"
            git config user.email "41898282+github-actions[bot]@users.noreply.github.com"

            echo "⚠️ final1.json یا normal1.json خالی یا موجود نیست، commit نشد"
          fi
